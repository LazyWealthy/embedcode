    <script>
        let container = document.getElementById("container");
        
        function reloadImage(img) {
            setTimeout(() => {
                img.src = img.src + "?retry=" + new Date().getTime();
            }, 1000);
        }

        function noColor(){
            container.style.background = "#ffffff";
        }

        const isMobile = /iPhone|Android/i.test(navigator.userAgent) && !/iPad|Tablet/i.test(navigator.userAgent);
        if (!isMobile) {
            
            container.style.width = "${dWidth}";
            container.style.height = "${dHeight}";
            container.style.marginLeft = "${dMargin-left}";
            document.getElementById("noteBlock").style.display = "flex";
            document.getElementById("noteText").style.display = "flex";
        }

    </script>
