    <script>
        const container = document.getElementById("container");
        const placeholder = document.getElementById("placeholder");
        const noteBlock = document.getElementById("noteBlock");
        const noteText = document.getElementById("noteText");
        const isMobile = /iPhone|Android/i.test(navigator.userAgent) && !/iPad|Tablet/i.test(navigator.userAgent) || window.innerWidth <= 330;
        const langCode = "en"
        const noteLabel = langCode == "zh" ? "æ³¨:" : "Note:";
        const dFootnote = "[$dFootnote]";
        const mFootnote = "[$mFootnote]";
        const footnote = isMobile ? mFootnote : dFootnote;

        function loadFooter() {

            if (footnote) {

                if (!isMobile) {
                    noteBlock.style.cssText = "display: flex; width: calc(100% - 30px); height: auto; min-height: 30px; margin-left: 15px; background: #fff; color: #000; font-family: 'Open Sans', sans-serif; padding-top: 15px; justify-content: left; font-size: 14px;";
                    noteText.innerHTML = noteLabel;
                    footer.innerHTML += dFootnote;
                } else {
                    noteBlock.style.cssText = "display: flex; width: calc(100% - 30px); height: auto; min-height: 30px; margin-left: 15px; background: #fff; color: #000; font-family: 'Open Sans', sans-serif; padding-top: 10px; justify-content: left; font-size: 12px;";
                    noteText.innerHTML = noteLabel;
                    footer.innerHTML += mFootnote;
                }

            }

        }

        let retryCount = 0;
        const maxRetries = 3;

        function reloadImage(img) {
            if (retryCount++ < maxRetries) {
                setTimeout(() => img.src += `?retry=${Date.now()}`, 1000);
            }
        }

        function clearBg() {
            container.style.removeProperty("min-height");
            container.style.background = "#ffffff";
            loadFooter();
        }

    </script>
