            try {
                const response = await fetch("https://raw.githubusercontent.com/LazyWealthy/embedcode/refs/heads/main/image.txt");
                if (!response.ok) throw new Error(`HTTP error! Status: ${response.status}`);

                let updatedScript = await response.text();
                const param = {
                    "[$imgsrc]": "https://1drv.ms/i/c/cda7c9b49e89a535/IQREdCukxqzISrLERTbiT0LvAdZRkKFTXvBEYppAFY9z7bY",
                    "[$dFootnote]": "",
                    "[$mFootnote]": "",
                    "[$langCode]": "en"
                };
                Object.entries(param).forEach(([search, replace]) => updatedScript = updatedScript.replaceAll(search, replace));
                // console.log("updatedScript", updatedScript);
                document.getElementById("container").innerHTML = updatedScript;
                document.querySelectorAll("#container script").forEach(script => {
                    const newScript = document.createElement("script");
                    newScript.textContent = script.textContent;
                    document.body.appendChild(newScript);
                });
            } catch (error) {
                console.error("Error loading or executing script:", error);
            }
        }

        fetchAndExecuteScript();
    </script>

<body>
    <div id="container">
        <img id="placeholder" style="width: 100%; height: 100%;" src="[$imgsrc]" onerror="reloadImage(this)"
            onload="clearBg()">
    </div>
    <div id="noteBlock" style="display: none;">
        <b id="noteText"></b>&nbsp;<span id="footer"></span>
    </div>

    <style>
        html,
        body {
            margin: 0;
            padding: 0;
            overflow: hidden;
        }

        #noteText {
            text-indent: 0.5em;
            display: inline-block;
        }
    </style>

    <script>
        const container = document.getElementById("container");
        const placeholder = document.getElementById("placeholder");
        const noteBlock = document.getElementById("noteBlock");
        const noteText = document.getElementById("noteText");
        const isMobile = /iPhone|Android/i.test(navigator.userAgent) && !/iPad|Tablet/i.test(navigator.userAgent) || window.innerWidth <= 330;
        const langCode = "en"
        const noteLabel = langCode == "zh" ? "æ³¨:" : "Note:";
        const dFootnote = "[$dFootnote]";
        const mFootnote = "[$mFootnote]";
        const footnote = isMobile ? mFootnote : dFootnote;

        function loadFooter() {

            if (footnote) {

                if (!isMobile) {
                    noteBlock.style.cssText = "display: flex; width: calc(100% - 30px); height: auto; min-height: 30px; margin-left: 0px; background: #fff; color: #000; font-family: 'Open Sans', sans-serif; align-items: center; justify-content: left; font-size: 14px;";
                    noteText.innerHTML = noteLabel;
                    footer.innerHTML += dFootnote;
                } else {
                    noteBlock.style.cssText = "display: flex; width: calc(100% - 30px); height: auto; min-height: 30px; margin-left: 0px; background: #fff; color: #000; font-family: 'Open Sans', sans-serif; align-items: center; justify-content: left; font-size: 12px;";
                    noteText.innerHTML = noteLabel;
                    footer.innerHTML += mFootnote;
                }

            }

        }

        let retryCount = 0;
        const maxRetries = 3;

        function reloadImage(img) {
            if (retryCount++ < maxRetries) {
                setTimeout(() => img.src += `?retry=${Date.now()}`, 1000);
            }
        }

        function clearBg() {
            container.style.removeProperty("min-height");
            container.style.background = "#ffffff";
            loadFooter();
        }

    </script>
</body>
