<body>
    <div id="container">
        <img id="placeholder" width="100%" height="100%" src="[$imgsrc]" onerror="reloadImage(this)"
            onload="loadFooter()">
        <div id="noteBlock"
            style="display: none; width: 100%; height: 60px; position: absolute; bottom: 0; z-index: 999; background: #fff; 
            color: #000; font-family: 'Open Sans', sans-serif; font-size: 14px; align-items: center; justify-content: left;">
            <b id="noteText">Note:</b>&nbsp;[$footnote]
        </div>
    </div>
    <style>
        html,
        body {
            margin: 0;
            padding: 0;
            overflow: hidden;
        }

        #noteText {
            text-indent: 1em;
            display: inline-block;
        }
    </style>

    <script>
        const container = document.getElementById("container");
        const noteBlock = document.getElementById("noteBlock");
        const noteText = document.getElementById("noteText");
        const isMobile = /iPhone|Android/i.test(navigator.userAgent) && !/iPad|Tablet/i.test(navigator.userAgent);
        const footnote = "[$footnote]";

        if (!isMobile) {
            container.style.width = "[$dWidth]px";
            container.style.height = "[$dHeight]px";
            container.style.marginLeft = "[$dMargin-left]px";
        }

        function loadFooter() {
            container.style.background = "#ffffff";
            const totalHeight = isMobile ? ([$mHeight] + 30) : ([$dHeight] + 60);
            console.log("totalHeight", totalHeight);
            if (footnote) {
                container.style.height = totalHeight.toString() + "px";
                container.style.height = "500px";
                noteBlock.style.display = "flex";
                noteText.style.display = "flex";
            }
        }

        function reloadImage(img) {
            setTimeout(() => {
                img.src = img.src + "?retry=" + new Date().getTime();
            }, 1000);
        }

    </script>
</body>
