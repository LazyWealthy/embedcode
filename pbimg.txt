<body>
    <div id="container">
        <img id="placeholder" style="display: none; width: 100%; height: 100%;" onerror="reloadImage(this)">
        <iframe id="dashboard" style="display: none;" frameborder="0" allowfullscreen>
        </iframe>
    </div>
    <div id="noteBlock" style="display: none;">
        <b id="noteText"></b>&nbsp;<span id="footer"></span>
    </div>

    <style>
        html,
        body {
            margin: 0;
            padding: 0;
            overflow: hidden;
        }

        #noteText {
            text-indent: 0.5em;
            display: inline-block;
        }
    </style>

    <script>
        const container = document.getElementById("container");
        const placeholder = document.getElementById("placeholder");
        const dashboard = document.getElementById("dashboard");
        const noteBlock = document.getElementById("noteBlock");
        const noteText = document.getElementById("noteText");
        const isMobile = /iPhone|Android/i.test(navigator.userAgent) && !/iPad|Tablet/i.test(navigator.userAgent) || window.innerWidth <= 330;
        const dashSrc = "[$dashsrc]";
        const mobImg = "[$imgsrc]";
        const langCode = "[$langCode]"
        const noteLabel = langCode == "zh" ? "æ³¨:" : "Note:";
        const dFootnote = "[$dFootnote]";
        const mFootnote = "[$mFootnote]";
        const footnote = isMobile ? mFootnote : dFootnote;

        if (!isMobile) {
            dashboard.src = dashSrc;
            dashboard.style.cssText = "display: block; width: [$dWidth]px; height: [$dHeight]px";
            dashboard.addEventListener("load", loadFooter());
        } else {
            placeholder.src = mobImg;
            placeholder.style.display = "block";
            placeholder.addEventListener("load", () => { loadFooter(); clearBg() });
        }

        function loadFooter() {

            container.style.removeProperty("min-height");

            if (footnote) {

                if (!isMobile) {
                    noteBlock.style.cssText = "display: flex; width: calc(100% - 30px); height: auto; min-height: 60px; margin-left: 0px; background: #fff; color: #000; font-family: 'Open Sans', sans-serif; align-items: center; justify-content: left; font-size: 14px; position: absolute; bottom: 0; z-index: 999;";
                    noteText.innerHTML = noteLabel;
                    footer.innerHTML += dFootnote;
                } else {
                    noteBlock.style.cssText = "display: flex; width: calc(100% - 30px); height: auto; min-height: 30px; margin-left: 0px; background: #fff; color: #000; font-family: 'Open Sans', sans-serif; align-items: center; justify-content: left; font-size: 12px; overflow: visible;";
                    noteText.innerHTML = noteLabel;
                    footer.innerHTML += mFootnote;
                }

            }

        }

        let retryCount = 0;
        const maxRetries = 3;

        function reloadImage(img) {
            if (retryCount++ < maxRetries) {
                setTimeout(() => img.src += `?retry=${Date.now()}`, 1000);
            }
        }

        function clearBg() {
            container.style.background = "#ffffff";
        }

    </script>
</body>
