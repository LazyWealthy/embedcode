<body>
    <div id="container"
        style="display: absolute; overflow:hidden; display:flex; justify-content:center; align-items:center;">
        <img id="placeholder" width="100%" height="100%" src="[$imgsrc]" onerror="reloadImage(this)"
            onload="loadFooter()">
        <iframe id="dashboard" display="none" width="[$dWidth]" height="[$dHeight]" src="[$dashsrc]" frameborder="0"
            allowfullscreen></iframe>
        <div id="noteBlock" style="display: none; width: 100%; position: absolute; bottom: 0; z-index: 999; background: #fff; 
            color: #000; font-family: 'Open Sans', sans-serif; align-items: center; justify-content: left;">
            <b id="noteText">Note:</b>&nbsp;<span id="footer"></span>
        </div>
    </div>
    <style>
        html,
        body {
            margin: 0;
            padding: 0;
            overflow: hidden;
        }

        #noteText {
            text-indent: 1em;
            display: inline-block;
        }
    </style>

    <script>
        const container = document.getElementById("container");
        const placeholder = document.getElementById("placeholder");
        const dashboard = document.getElementById("dashboard");
        const noteBlock = document.getElementById("noteBlock");
        const noteText = document.getElementById("noteText");
        const isMobile = /iPhone|Android/i.test(navigator.userAgent) && !/iPad|Tablet/i.test(navigator.userAgent);
        const dFootnote = "[$dFootnote]";
        const mFootnote = "[$mFootnote]";
        const footnote = isMobile ? mFootnote : dFootnote;

        if (!isMobile) {
            container.style.width = "[$dWidth]px";
            container.style.marginLeft = "[$dMargin-left]px";
            placeholder.style.cssText = "width: 1px; height: 1px; position: absolute;";
            dashboard.style.display = "block";
        } else {
            noteBlock.style.marginLeft = "[$mMargin-left]px";
            dashboard.style.display = "none";
        }

        function loadFooter() {
            container.style.background = "#ffffff";
            if (footnote) {
                const noteFont = isMobile ? 12 : 14;
                const noteHeight = isMobile ? 30 : 60;
                const totalHeight = isMobile ? ([$mHeight] + noteHeight) : ([$dHeight]);
                container.style.height = totalHeight + "px";
                noteBlock.style.height = noteHeight + "px";
                noteBlock.style.fontSize = noteFont + "px";
                noteBlock.style.display = "flex";
                noteText.style.display = "flex";
                footer.innerHTML += " " + footnote;
            } else {
                const normalHeight = isMobile ? ([$mHeight]) : ([$dHeight]);
                container.style.height = normalHeight + "px";
            }
        }

        let retryCount = 0;
        const maxRetries = 3;

        function reloadImage(img) {
            if (retryCount++ < maxRetries) {
                setTimeout(() => img.src += `?retry=${Date.now()}`, 1000);
            }
        }

    </script>
</body>
